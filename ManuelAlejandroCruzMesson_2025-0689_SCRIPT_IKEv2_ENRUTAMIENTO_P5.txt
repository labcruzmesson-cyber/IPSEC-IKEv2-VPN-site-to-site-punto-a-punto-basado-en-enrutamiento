configure terminal

crypto ikev2 proposal PROP_IKEV2
 encryption aes-cbc-256
 integrity sha256
 group 14
exit

crypto ikev2 policy POL_IKEV2
 match fvrf any
 proposal PROP_IKEV2
exit

crypto ikev2 keyring KEYRING_VPN
 peer PEER_B
  address 10.0.0.70
  pre-shared-key cisco123!
 exit
exit

crypto ikev2 profile PROFILE_IKEV2
 match identity remote address 10.0.0.70 255.255.255.255
 identity local address 10.0.0.60
 authentication remote pre-share
 authentication local pre-share
 keyring local KEYRING_VPN
exit

crypto ipsec transform-set TS_VPN esp-aes 256 esp-sha256-hmac
 mode tunnel
exit

crypto ipsec profile IPSEC_PROFILE
 set transform-set TS_VPN
 set ikev2-profile PROFILE_IKEV2
exit

interface Tunnel0
 ip address 192.168.100.1 255.255.255.252
 tunnel source GigabitEthernet0/0
 tunnel destination 10.0.0.70
 tunnel mode ipsec ipv4
 tunnel protection ipsec profile IPSEC_PROFILE
no shutdown
exit

no ip route 172.16.2.0 255.255.255.0 10.0.0.70
ip route 172.16.2.0 255.255.255.0 Tunnel0

ip access-list extended ACL_NAT_INTERNET
 no permit ip 172.16.1.0 0.0.0.255 any
 10 deny ip 172.16.1.0 0.0.0.255 172.16.2.0 0.0.0.255
 20 permit ip 172.16.1.0 0.0.0.255 any
exit
